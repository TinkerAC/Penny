项目根目录

│       │   │           │   │   ├── TransactionRepository.kt    // 交易数据仓库类
│       │   │           │   │   └── ...                 // 其他仓库类

│       │   │           │       └── ...                 // 其他模型类
│       │   │           ├── domain                      // 领域层：业务逻辑和模型

│       │   │           │       │   ├── TransactionScreen.kt    // 交易列表界面
